Dropdown

A dropdown combobox is an input widget that, when activated, displays a popup listbox containing predefined options. The user can select one option from the list. It does not allow freeform text entry — see Dropdown with search for that variant.

Accessibility Features

  • Opens and closes with Space, Enter, and arrow keys
  • Options are navigable with ↑ and ↓ arrow keys
  • Pressing Escape closes the dropdown and returns focus to the trigger
  • The expanded/collapsed state is conveyed with aria-expanded attribute
  • The widget role is conveyed with role="combobox" on the trigger element

Example

Code Snippet

<div class="combobox">
  <button
    id="combobox-trigger"
    role="combobox"
    aria-haspopup="listbox"
    aria-expanded="false"
    aria-controls="combobox-listbox"
  >
    <span id="combobox-label">Select an option</span>
    <span aria-hidden="true">▼</span>
  </button>
  <ul
    id="combobox-listbox"
    role="listbox"
    aria-labelledby="combobox-label"
    hidden
  >
    <li role="option" aria-selected="false">Option 1</li>
    <li role="option" aria-selected="false">Option 2</li>
    <li role="option" aria-selected="false">Option 3</li>
  </ul>
</div>

const trigger = document.getElementById('combobox-trigger');
const listbox = document.getElementById('combobox-listbox');
const options = listbox.querySelectorAll('[role="option"]');
let activeIndex = -1;

trigger.addEventListener('click', toggleDropdown);

trigger.addEventListener('keydown', (e) => {
  if (e.key === 'ArrowDown' || e.key === 'Enter' || e.key === ' ') {
    e.preventDefault();
    openDropdown();
    focusOption(0);
  }
});

listbox.addEventListener('keydown', (e) => {
  if (e.key === 'ArrowDown') {
    e.preventDefault();
    focusOption(Math.min(activeIndex + 1, options.length - 1));
  } else if (e.key === 'ArrowUp') {
    e.preventDefault();
    focusOption(Math.max(activeIndex - 1, 0));
  } else if (e.key === 'Enter') {
    selectOption(options[activeIndex]);
  } else if (e.key === 'Escape') {
    closeDropdown();
  }
});

options.forEach((option) => {
  option.addEventListener('click', () => selectOption(option));
});

function toggleDropdown() {
  if (listbox.hidden) openDropdown(); else closeDropdown();
}

function openDropdown() {
  listbox.hidden = false;
  trigger.setAttribute('aria-expanded', 'true');
}

function closeDropdown() {
  listbox.hidden = true;
  trigger.setAttribute('aria-expanded', 'false');
  activeIndex = -1;
  trigger.focus();
}

function focusOption(index) {
  if (activeIndex >= 0) options[activeIndex].removeAttribute('tabindex');
  activeIndex = index;
  options[index].setAttribute('tabindex', '0');
  options[index].focus();
}

function selectOption(option) {
  options.forEach(o => o.setAttribute('aria-selected', 'false'));
  option.setAttribute('aria-selected', 'true');
  document.getElementById('combobox-label').textContent = option.textContent;
  closeDropdown();
}
